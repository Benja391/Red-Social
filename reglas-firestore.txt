      allow write: if request.auth != null && 
                    (request.auth.uid == resource.data.userId || request.auth.token.admin == true);
    }

    // Reglas para la colecci贸n de comentarios

      allow write: if request.auth != null && 
                    (request.auth.uid == resource.data.userId || request.auth.token.admin == true);
    }
  }
